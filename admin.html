export default { async fetch(request, env, ctx) { return new Response("TON Worker Running ✅"); }, async scheduled(event, env, ctx) { console.log("Starting TON deposit check..."); const wallet = env.WALLET_ADDRESS; // 🔹 جلب آخر المعاملات من TON API const response = await fetch( `https://toncenter.com/api/v2/getTransactions?address=${wallet}&limit=30`, { headers: { "X-API-Key": env.TON_API_KEY } } ); const data = await response.json(); if (!data.result) { console.log("No transactions found"); return; } const transactions = data.result; for (const tx of transactions) { const txHash = tx.transaction_id.hash; if (!tx.in_msg || !tx.in_msg.message) continue; let comment = tx.in_msg.message; if (!comment) continue; comment = comment.trim(); // 🔹 نتأكد إن الكومنت رقم فقط if (!/^\d+$/.test(comment)) continue; const userId = comment; const amountNano = tx.in_msg.value; const amountTon = Number(amountNano) / 1e9; if (amountTon <= 0) continue; console.log(`Processing deposit for user ${userId}`); // 🔹 تحقق هل المعاملة اتعالجت قبل كده const checkProcessed = await fetch( `${env.FIREBASE_DB_URL}/processed/${txHash}.json` ); const alreadyProcessed = await checkProcessed.json(); if (alreadyProcessed) { console.log("Transaction already processed"); continue; } // 🔹 جلب رصيد المستخدم الحالي const userRes = await fetch( `${env.FIREBASE_DB_URL}/users/${userId}.json` ); const userData = await userRes.json(); if (!userData) continue; const currentBalance = userData.tonBalance || 0; const newBalance = currentBalance + amountTon; // 🔹 تحديث الرصيد await fetch(`${env.FIREBASE_DB_URL}/users/${userId}.json`, { method: "PATCH", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ tonBalance: newBalance }) }); // 🔹 تعليم المعاملة كمُعالجة await fetch(`${env.FIREBASE_DB_URL}/processed/${txHash}.json`, { method: "PUT", headers: { "Content-Type": "application/json" }, body: JSON.stringify(true) }); console.log(`Added ${amountTon} TON to user ${userId}`); // ========================= // 🔔 User Message (English) // ========================= const formattedAmount = amountTon.toFixed(18); const walletLink = `https://tonviewer.com/${wallet}`; const userMessage = ` ✅ Deposit Confirmed! Your deposit has been credited successfully! 🎉 ━━━━━━━━━━━━━━━━ 💰 Amount: ${formattedAmount} TON 💼 Receiving Wallet: Crystal Ranch ━━━━━━━━━━━━━━━━ Thank you for your trust! ✨ Have a great day! 🌟 `; await fetch(`https://api.telegram.org/bot${env.BOT_TOKEN}/sendMessage`, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ chat_id: userId, text: userMessage, parse_mode: "HTML", disable_web_page_preview: true }) }); console.log("Telegram notification sent to user"); // ========================= // 📢 Group Message - مع تحديد الموضوع الصحيح (Withdrawals & deposit $) // ========================= const groupMessage = ` 🎉 NEW DEPOSIT JUST ARRIVED! 🔥 ━━━━━━━━━━━━━━━━ 👤 Username: ${userData.username || 'Unknown'} 🆔 User ID: ${userId} 💰 Amount: ${formattedAmount} TON 💼 Wallet: Crystal Ranch ━━━━━━━━━━━━━━━━ `; // إرسال إلى المجموعة في الموضوع المحدد (Topic ID: 5) await fetch(`https://api.telegram.org/bot${env.BOT_TOKEN}/sendMessage`, { method: "POST", headers: { "Content-Type": "application/json" }, body: JSON.stringify({ chat_id: "@Crystal_Ranch_chat", // المجموعة الرئيسية message_thread_id: 5, // معرف الموضوع "Withdrawals & deposit $" text: groupMessage, parse_mode: "HTML", disable_web_page_preview: false }) }); console.log("Group notification sent to Withdrawals & deposit $ topic"); } console.log("Deposit check completed."); } };
